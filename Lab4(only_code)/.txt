import { useState } from 'react'
const ToDoFunction =()=>{
  const [tasks ,setTask]=useState([]);
  const [taskInput,setTaskInput]=useState('');
  const addTask=()=>{
    if(taskInput.trim()!==''){
      setTask([...tasks,{text:taskInput,completed:false}]);
      setTaskInput('');
    }
  };
  //toggle task complted
  const toggleTask=(index)=>{
    const updatedTasks=[...tasks];
    updatedTasks[index].completed=!updatedTasks[index].completed;
    setTask(updatedTasks);
  };
//delete task
const deleteTask=(index)=>{
  const updatedTask=tasks.filter((_,i)=>i !==index);
  setTask(updatedTask);
};
 return(
    <div style={styles .container}>
      <h1>React to-Do List</h1>
      <div>
        <input 
        type="text"
        value={taskInput}
        onChange={(e)=>setTaskInput(e.target.value)}
        placeholder="Enter a task..."
        style={styles.input}
      />
      <button onClick={addTask} style={styles.addButton}>Add Task</button>
      </div>
      <ul style={styles .list}>
        {tasks.map((task,index)=>(
          <li key={index} style={task.completed ? styles.completedTask : styles.pendingTask}>
            <span onClick={()=>toggleTask(index)} style={{ cursor:'pointer'}}>
              {index+1}.{task.completed ?'✅':'❌'} {task.text}
            </span>
            <button onClick={()=>deleteTask(index)} style={styles.deleteButton}>🗑️</button>
          </li>
        ))}
      </ul>
    </div>
  );
};
  const styles ={
  container:{
    textAlign:'center',
    fontFamily:'Arial,sans-serif',
    marginTop:'50px',
  },
  input:{
    padding:'10px',
    fontSize:'16px',
    width:'250px',
  },
  addButton:{
    marginLeft:'10px',
    padding:'10px',
    cursor:'pointer',
  },
  list:{
    listStyleType:'none',
    padding:0,
    marginTop:'20px',
  },
  pendingTask:{
    padding:'10px',
    fontSize:'18px',
    borderBottom:'1px solid #ddd',
    display:'Flex',
    justifyContent:'space-between',
    alignItems:'center',
  },
  completedTask:{
    padding:'10px',
    fontSize:'18px',
    textDecoration:'line-through',
    color:'gray',
    borderBottom:'1px solid #ddd',
    display:'Flex',
    justifyContent:'space-between',
    alignItems:'center',
  },
  deleteButton:{
    background:'none',
    border:'none',
    fontSize:'18px',
    cursor:'pointer',
  },
};
export default ToDoFunction;
